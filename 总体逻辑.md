5️⃣ 每日闭环流程
每日流程：
1. 玩家登录 → 获得基础香火钱
2. agent 自动执行结缘任务 → 获得修行奖励
3. 随机触发特殊事件 → 信物赠送（增加经验）
4. 玩家可花香火钱操作（点香/供奉/添香） → 增加经验
5. 累计经验 → 升级
6. 每天日志显示：几点到寺庙、结缘、祈福、获得信物
7. 第五天完成后等级 >= 12 → 解锁所有 12 个寺庙
6️⃣ Python 示例实现
class Monk:
    def __init__(self):
        self.level = 1
        self.exp = 0
        self.base_exp = 100
        self.growth_rate = 1.5
        self.incense_coin = 0
        self.unlocked_temples = ["寺庙1"]
        self.temples_total = 12

    def exp_to_next_level(self):
        return int(self.base_exp * (self.level ** self.growth_rate))

    def add_exp(self, gained_exp):
        self.exp += gained_exp
        while self.exp >= self.exp_to_next_level():
            self.exp -= self.exp_to_next_level()
            self.level_up()

    def level_up(self):
        self.level += 1
        if self.level <= self.temples_total:
            new_temple = f"寺庙{self.level}"
            self.unlocked_temples.append(new_temple)
            print(f"升到等级 {self.level}! 解锁 {new_temple}")

    def daily_login(self, day):
        login_rewards = [10,12,14,16,18]  # 天1~5
        task_rewards = [20,22,24,26,28]
        special_rewards = [5,5,5,5,5]      # 随机可触发
        self.incense_coin += login_rewards[day-1] + task_rewards[day-1]
        # 随机特殊事件
        import random
        if random.random() < 0.3:
            self.incense_coin += special_rewards[day-1]
            print("特殊事件！获得道友信物，香火钱增加")
        print(f"第{day}天香火钱: {self.incense_coin}")

    def use_incense_coin(self, action):
        # 点香/供奉/添香
        if action == "点香" and self.incense_coin >= 1:
            self.incense_coin -= 1
            self.add_exp(5)
        elif action == "供奉" and self.incense_coin >= 5:
            self.incense_coin -= 5
            self.add_exp(30)
        elif action == "添香" and self.incense_coin >= 10:
            self.incense_coin -= 10
            self.add_exp(70)
        else:
            print("香火钱不足或操作错误")
